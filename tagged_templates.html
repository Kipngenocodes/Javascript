How This Works

This demo shows how tagged template literals work in JavaScript. The format function receives:

  • The string parts (split by the expressions)
  • The evaluated expressions as individual arguments

The function then combines these in a custom way to create the final string.

The Code

function format(strings, ...values) {
  // Log information about inputs
  console.log("String parts:", strings);
  console.log("Expression values:", values);
  
  // Display the raw values for demonstration
  document.getElementById("output1").innerHTML = 
    "Name: " + values[0] + ", Price: " + values[1];
  
  // Combine strings and values in alternating order
  let result = strings[0];
  for (let i = 0; i < values.length; i++) {
    result += values[i] + strings[i + 1];
  }
  
  return result;
}

const price = 100;
let name = "John";
const res = format`Hi, ${name}! The price is $${price}.`;
document.getElementById("output2").innerHTML = res;
